use std::env;
use std::fs;
use std::io::Write;
use std::path::PathBuf;
use std::process::{Command, Stdio};
use std::time::{SystemTime, UNIX_EPOCH};

fn main() -> std::io::Result<()> {
    // ------------------------------------------------------------
    // 1. Build a unique name for the temporary executable
    // ------------------------------------------------------------
    // `temp_dir()` is usually `/tmp` on Unix-like systems.
    let temp_dir = env::temp_dir();

    // Compose a file name that is very unlikely to collide:
    //   <process‑id>_<timestamp‑in‑nanoseconds>
    let pid = std::process::id();
    let ts = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .expect("Time went backwards")
        .as_nanos();
    let exe_name = format!("hello_{}_{}.exe", pid, ts);

    let exe_path: PathBuf = temp_dir.join(exe_name);

    // ------------------------------------------------------------
    // 2. Compile the C source (sent via stdin)
    // ------------------------------------------------------------
    // The C program that prints “Hello, World!”.
    let c_source = r#"#include <stdio.h>
        int main(){
            printf("Hello, World!\n"); 
            return 0;
        }"#;

    // `cc` is a very common C compiler driver (it might be a symlink to
    // `gcc`, `clang`, …).  The arguments mean:
    //   -x c      : treat the input as C code
    //   -         : read source from stdin
    //   -o <file> : write the executable to <file>
    let mut compiler = Command::new("cc")
        .args(&["-x", "c", "-"])
        .arg("-o")
        .arg(&exe_path)
        .stdin(Stdio::piped())
        .stderr(Stdio::inherit())   // show compile errors if any
        .spawn()?;                  // start the compiler

    // Write the source to the compiler's stdin.
    if let Some(mut stdin) = compiler.stdin.take() {
        stdin.write_all(c_source.as_bytes())?;
    }

    // Wait for the compiler to finish.
    let status = compiler.wait()?;
    if !status.success() {
        eprintln!("Compilation failed.");
        std::process::exit(1);
    }

    // ------------------------------------------------------------
    // 3. Run the compiled binary
    // ------------------------------------------------------------
    let output = Command::new(&exe_path)
        .output()
        .expect("Failed to execute the compiled program");

    // Print the stdout of the child process.
    let stdout_text = String::from_utf8_lossy(&output.stdout);
    print!("{}", stdout_text);

    // (Optional) If you want to see any error output from the program:
    let stderr_text = String::from_utf8_lossy(&output.stderr);
    if !stderr_text.is_empty() {
        eprintln!("stderr: {}", stderr_text);
    }

    // ------------------------------------------------------------
    // 4. Clean up: delete the temporary executable
    // ------------------------------------------------------------
    fs::remove_file(&exe_path)?;

    Ok(())
}
