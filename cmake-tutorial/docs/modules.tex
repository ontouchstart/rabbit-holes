\section{Tutorial Modules - Source Code}

\subsection{Add Executable Module}
This module demonstrates a simple executable returning an exit code.

\begin{lstlisting}[style=cppstyle, caption=src/main.cc]
int main () { return 42; }
\end{lstlisting}

\subsection{Add Library Module}
This module demonstrates static library creation.

\begin{lstlisting}[style=cppstyle, caption=include/42.h]
#ifndef ANSWER
#define ANSWER 42
#endif
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption=include/answer.h]
#ifndef ANSWER_H
int answer();
#endif
\end{lstlisting}

\begin{lstlisting}[style=cppstyle, caption=lib/answer.cc]
#include "answer.h"
#include "42.h"
int answer () { return ANSWER; }
\end{lstlisting}

\subsection{Target Link Libraries Module}
This module demonstrates linking an executable against a static library.

\begin{lstlisting}[style=cppstyle, caption=src/main.cc (modified)]
#include "answer.h"
int main () { return answer(); }
\end{lstlisting}
